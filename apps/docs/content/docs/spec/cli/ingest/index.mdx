## ingest submit

<CommandSpec
  command="ingest submit"
  mutates={true}
  summary="파싱된 거래를 적재한다. 기본 draft. --post면 posted. OCR 없음."
  signature={`holiday ingest submit
  (--data <json> | --data-file <path>)
  [--source|--image <path>] [--idem-key <key>] [--post]`}
  flags={[
    { name: '--data', type: 'json', description: '인라인 submission JSON. --data-file과 정확히 하나' },
    { name: '--data-file', type: 'path', description: '파일에서 JSON 읽기 (대량 수입)' },
    { name: '--source', type: 'path', description: '원본 파일 해시. 동일 바이트 재수입 차단' },
    { name: '--image', type: 'path', description: '--source 별칭 (캡쳐용)' },
    { name: '--idem-key', type: 'string', description: '동일 키+요청이면 저장된 응답 재생' },
    { name: '--post', type: 'boolean', default: 'false', description: 'review 건너뛰고 posted로 커밋' },
  ]}
  exits={[
    { code: 0, meaning: '배치 적재(또는 idempotent replay)' },
    { code: 1, meaning: 'duplicate_image / duplicate_external_ref / unbalanced' },
    { code: 2, meaning: 'JSON·계정·idem·data/data-file 오류' },
  ]}
  example={`holiday ingest submit --data-file ./batch.json --source ./export.csv --idem-key run-1 --json`}
/>
